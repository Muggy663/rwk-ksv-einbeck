// Firestore Security Rules für KM-Collections
// Diese Rules müssen in Firebase Console → Firestore → Rules hinzugefügt werden

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // KM Meldungen - Schreibzugriff für authentifizierte Benutzer
    match /km_meldungen/{document} {
      allow read, write: if request.auth != null;
    }
    
    // KM Disziplinen - Lesezugriff für alle, Schreibzugriff für Admins
    match /km_disziplinen/{document} {
      allow read: if true;
      allow write: if request.auth != null;
    }
    
    // KM Wettkampfklassen - Lesezugriff für alle, Schreibzugriff für Admins  
    match /km_wettkampfklassen/{document} {
      allow read: if true;
      allow write: if request.auth != null;
    }
    
    // Bestehende Rules für andere Collections bleiben unverändert...
  }
}
